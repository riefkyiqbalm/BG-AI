# ISA constants
_RHO0 = 1.225    # kg/m3  sea level density
_T0   = 288.15   # K      sea level temperature
_L    = 0.0065   # K/m    lapse rate
_G    = 9.81     # m/s2
_R    = 287.05   # J/(kg?K)
_MU   = 1.789e-5 # Pa?s


def _isa_density(altitude_m: float) -> float:
    """Return ISA air density at given altitude (m)."""
    T   = _T0 - _L * altitude_m
    p   = 101325.0 * (T / _T0) ** (_G / (_L * _R))
    rho = p / (_R * T)
    return rho
